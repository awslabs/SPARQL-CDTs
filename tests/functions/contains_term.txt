PREFIX cdt:    <http://example.org/cdt/>
PREFIX xsd:    <http://www.w3.org/2001/XMLSchema#>

# empty lists
ASK {
	BIND("[]"^^cdt:List AS ?list)
	FILTER(!cdt:containsTerm(?list, 1))
};

# contains integer
ASK {
	BIND("[1]"^^cdt:List AS ?list)
	FILTER(cdt:containsTerm(?list, 1))
	FILTER(!cdt:containsTerm(?list, 2))
};

# contains equality
ASK {
	BIND("[1,'a','b'@en,2.0]"^^cdt:List AS ?list)

	# numeric forms
	FILTER(cdt:containsTerm(?list, 1))
	FILTER(!cdt:containsTerm(?list, "+1"^^xsd:integer))
	FILTER(!cdt:containsTerm(?list, "01"^^xsd:integer))
	FILTER(!cdt:containsTerm(?list, 1.0))
	FILTER(!cdt:containsTerm(?list, 1.0))
	FILTER(!cdt:containsTerm(?list, 1e0))
	FILTER(!cdt:containsTerm(?list, 2))
	FILTER(cdt:containsTerm(?list, 2.0))
	FILTER(!cdt:containsTerm(?list, 2e0))
	
	# string forms
	FILTER(cdt:containsTerm(?list, 'a'))
	FILTER(cdt:containsTerm(?list, 'b'@en))
	FILTER(!cdt:containsTerm(?list, 'b'))
};
